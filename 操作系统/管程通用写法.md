[[并发程序设计]]
```c
monitor M{
	int state;//共享变量
	
	condition c1,c2;//条件变量
	
	proceduce operation1(){
		while(不满足 operation1 的执行条件){
			wait(c1);
		}
		
		修改共享状态
		
		signal(c2);
	}
	proceduce operation2(){
		while(不满足 operation2的执行条件){
			wait(c2);
		}
		
		修改共享状态
		
		signal(c1);
	}
}
```

读者-写者题目实例
允许同时读，写时禁止写和读
```c
monitor RW{
	bool writing=false;
	int reader=0;
	
	condition canRead;
	condition canWrite;
	
	procedure Write(){
		while(writing||reader>0){
			wait(canWrite);
		}
		
		writing = true;
		write()
		writing = false;
		
		signal(canRead);
		signal(canWrite);
	}
	
	procedure startRead(){
		while(writing){
			wait(canRead);
		}
		
		reader++;
		signal(canRead);
	}
	procedure endRead(){
		reader--;
		if(!reader){
			signal(canWrite);
		}
	}
}
```
